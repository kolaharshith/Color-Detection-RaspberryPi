<script>
async function fetchColor() {
    let res = await fetch('/color');
    let data = await res.json();

    document.getElementById("rgb").innerText =
        `R: ${data.r} | G: ${data.g} | B: ${data.b}`;

    document.getElementById("colorName").innerText =
        data.color.toUpperCase();

    document.getElementById("colorName").style.color =
        data.color;

    document.getElementById("colorBox").style.background =
        data.color;
}

setInterval(fetchColor, 1000);
</script>
